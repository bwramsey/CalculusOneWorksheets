% Extracted from derivativesOfInverseFunctions.tex, problem #1
\begin{problem}
Explain what each of the following means:
	\begin{enumerate}
	%part a
	\item  $\sin^{-1}(x)$
\WkstHop
		\begin{freeResponse}
		This denotes the inverse function to $\sin(x)$, sometimes denoted by $\arcsin(x)$.  
		\end{freeResponse}	
		
	%part b
	\item  $\left( \sin(x) \right)^{-1}$
\WkstHop
		\begin{freeResponse}
		This means $\sin(x)$ raised to the $-1$ power, i.e. $\frac{1}{\sin(x)}$.  
		\end{freeResponse}	
		
	%part c
	\item  $\sin \left(x^{-1} \right)$
\WkstHop
		\begin{freeResponse}
		This means $\sin \left( \frac{1}{x} \right)$.
		\end{freeResponse}
	\end{enumerate}
\end{problem}
