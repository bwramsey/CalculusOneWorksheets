\begin{problem}
Find the  x-coordinates of global extrema and global extreme values  of $f$  on the given interval.
		\begin{enumerate}
		
			%part a
			\item  $f(x) = x \sqrt{2-x^2}$ on $[ -\sqrt{2}, \sqrt{2} ]$.
%part b
			\item  $f(x) = x^3 e^{-x}$ on $[-1,5]$.
%part c
			\item  $f(x) = x \ln \left( \frac{x}{5} \right)$ on $[0.1, 5]$.
\end{enumerate}

\end{problem}
