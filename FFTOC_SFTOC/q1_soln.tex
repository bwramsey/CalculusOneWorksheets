\begin{problem}
  True of False: If $f$ is continuous on the closed interval $[a,b]$,
  then
  $$\ddx \left( \int_a^b f(t) \d t \right) = f(x)$$
  \begin{explanation}
    False.  $\ddx \left( \int_a^b f(t) \d t \right) = 0$
    since $\int_a^b f(t) \d t$ is a constant and
    $\ddx (\text{constant}) = 0$.
  \end{explanation}
\end{problem}
