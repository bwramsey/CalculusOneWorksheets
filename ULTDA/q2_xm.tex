\begin{problem}
	Sketch a possible graph of a function $g$ that satisfies the following conditions:
	\begin{align*}
	&\textrm{Domain:} [-5,-2) \cup (-2,3) \cup (3,5)
	& \lim_{x \to -2} g(x) = 3\\
	& g(1)=1
	& \lim_{x \to 1^+} g(x) = -\infty\\
	& \lim_{x \to 3} g(x) = \infty
	& \lim_{x \to 5^-} g(x) = \infty\\
	& \lim_{x \to 1^-} g(x) = 1
	& g(-5)=-1.8
	\end{align*}
\end{problem}
