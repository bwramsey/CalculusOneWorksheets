\begin{problem}

  Circle the correct answer in each part:
  \begin{enumerate}
    \item
      Consider the limit $\lim_{x \to 0} (\cos(x))^{\sin(x)}$.
      \begin{enumerate}
        \item
          Evaluate the limit.
         
          \begin{enumerate}
            \item
              the limit DNE
            \item
              $e$
             \item
              $1$
            \item
              $\infty$
            \item
              $-\infty$
            \item
              $0$
            \item
              none of the previous answers is correct
          \end{enumerate}
           \begin{explanation}
            \textbf{The correct choice is (iii).}

            Evaluation of limit:
            \begin{align*}
              \lim_{x \to 0} \underbrace{(\cos(x))^{\sin(x)}}_{\text{form $1^0$}} &= 1
            \end{align*}
          \end{explanation}
        \item
          What Limit Law, rule or technique did you use to find this limit?
                    \begin{enumerate}
            \item
              The Squeeze Theorem;
            \item
              L'H\^{o}pital's Rule;
            \item
              The Product Law;
            \item
              evaluated the function at $x = 0$, since the function is continuous at $x = 0$;
            \item
              none of the previous answers is correct
          \end{enumerate}
          \begin{explanation}
            The correct choice is (iv).
          \end{explanation}

      \end{enumerate}
\item
      Evaluate the limit $\lim_{x \to 4^-} \frac{\ln(x)}{x - 4}$.
  
      \begin{enumerate}
        \item
          the limit DNE
        \item
          $e$
        \item
          $1$
        \item
          $\infty$
        \item
          $-\infty$
        \item
          $0$
        \item
          none of the previous answers is correct
      \end{enumerate}
  \begin{explanation}
        The correct choice is (v).

        Evaluation of limit:
  \(
          \lim_{x \to 4^-}  \frac{\ln(x)}{x - 4}= - \infty, 
 \)
 \\[1em]
 \text{since the limit is of the form $\dfrac{\#}{0}$, and since  the numerator  is positive and the denominator negative.}
      \end{explanation}

    \item
      Evaluate the limit $\lim_{x \to \infty} \frac{\ln(x)}{x - 4}$.
           \begin{enumerate}
        \item
          the limit DNE
        \item
          $e$
        \item
          $1$
        \item
          $\infty$
        \item
          $-\infty$
        \item
          $0$
        \item
          none of the previous answers is correct
      \end{enumerate}
 \begin{explanation}
        The correct choice is (vi).

        Evaluation of limit:
        \begin{align*}
          \lim_{x \to \infty} \underbrace{\frac{\ln(x)}{x - 4}}_{\text{form $\infty/\infty$}} &\stackrel{L.H.}{=} \lim_{x \to \infty} \frac{1/x}{1} \\
          &= 0
        \end{align*}

      \end{explanation}

\item
      Consider the limit\\[1em]
       $\lim_{h \to 0} \frac{(2+h)^3 - 8}{h} = f'(2)$.\\[1em]
      Determine the function $f$.
    
      \begin{enumerate}
       \item
         such a function DNE;
       \item
         $f(x) = x^3$;
       \item
         $f(x) = (2 + x)^3$;
       \item
         $f(x) = \frac{(2+x)^3}{x}$;
       \item
         none of the previous answers is correct
      \end{enumerate}
        \begin{explanation}
        The correct choice is (ii).
      \end{explanation}
    \item
      Consider the limit: $\lim_{x \to 0^+} \left( \frac{\sin(x)}{x} \right)^{|\ln(x)|}$.\\[1em]
      Determine the form of this limit.
           \begin{enumerate}
       \item
         $\frac{0}{0}$;
       \item
         $\frac{\infty}{\infty}$;
       \item
         $1^0$;
       \item
         $0^0$;
       \item
         $1^\infty$;
       \item
         $\infty^\infty$;
       \item
         none of the previous answers is correct
      \end{enumerate}
       \begin{explanation}
        The correct choice is (v).
      \end{explanation}

  \end{enumerate}
\end{problem}
