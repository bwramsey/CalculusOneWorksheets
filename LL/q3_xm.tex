\begin{problem}
Suppose
	$f(x) =   \left\{ \begin{array}{lr}
	x^2 - ax 	&	\text{if } x < 3	\\ \\
	a 2^x + 7 + a	&	\text{if } x > 3	\end{array} \right.  $
	
	Find $a$ so that $ \lim_{x \to 3} f(x)$ exists. 
\end{problem}
