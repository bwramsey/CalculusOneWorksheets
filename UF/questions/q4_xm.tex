\begin{problem}
Given $y(t)=t- \frac{\pi}{3}$ and $w(t)=\sin(t)$.  Find:
\begin{enumerate}	
	\item  $y(w(t))$
		\item  $w(y(t))$
		\item  $w \left(y \left(\frac{4\pi}{3} \right)\right)$
		\item  $y(w(\frac{4\pi}{3}))$
		\end{enumerate}

\end{problem}
