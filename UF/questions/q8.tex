\begin{problem}

\begin{enumerate}
 	\item $y=\sqrt{x+2}-3$
	\WkstHop
	
	\begin{freeResponse}

		This is a shift of $y=\sqrt{x}$ moved left two units and down three units.
	
		\begin{image}		
			\includegraphics[scale = 0.5, alt={Graph of square root function and of it shifted 2 units down and 3 units to the left.}]{UFFigure8.png }

	\end{freeResponse}
	

	\item $y=\frac{1}{x-4}+1$
		\WkstHop

\end{enumerate}

\end{problem}
