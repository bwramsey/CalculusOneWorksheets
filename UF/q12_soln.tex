\begin{problem}
Explain what each of the following means:

  \begin{enumerate}

    % part a
  \item $f^{-1}(x)$
\begin{explanation}
      This denotes the inverse function of $f$, $f^{-1}$, evauluated at $x$.
    \end{explanation}
		
    % part b
  \item $f(x^{-1})$
\begin{explanation}
      This means $f \left( \frac{1}{x} \right)$.
    \end{explanation}
		
    % part c
  \item $\left( f(x) \right)^{-1}$
\begin{explanation}
      This means $f(x)$ raised to the $-1$ power,
      i.e. $\frac{1}{f(x)}$.
    \end{explanation}
  \end{enumerate}
\end{problem}
