\begin{problem}
Suppose the line tangent to the graph of $f$ at $x=1$ is $y=6x-7$.  Find an equation of the line tangent to the following curves at $x=1$:
	\begin{enumerate}
	
	\item  $y=g(x) = 5(f(x))^4$  
\item  $y=h(x) = x^2 (f(x^3))$
\end{enumerate}		

\end{problem}
