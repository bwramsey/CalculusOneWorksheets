\begin{problem} 
	 The graph of a function $f$ is given below.  Use this graph to answer the following questions.

	  \begin{image}
	    \includegraphics[scale = 0.3, alt={graph of piecewise-defined function for finding domain, range, and limit values from graph.}]{pwDefFunction.png}
	  \end{image}

 \begin{enumerate}
    \item
        Find the domain of $f$.
	\begin{explanation}
	          The domain of $f$ is $(-\infty, -2) \cup (-2, \infty)$.
	        \end{explanation}


    \item Find the range of $f$.
	\begin{explanation}
          	The range of $f$ is $(-\infty, \infty)$.
          \end{explanation}

    \item Find the following values.
      
      \begin{enumerate}
        \item $\displaystyle \lim_{x \to -2} f(x) = $
\begin{explanation}
	            $\displaystyle \lim_{x \to -2} f(x) = 0$
          \end{explanation}

        \item $f(-2) = $
\begin{explanation}
            $f(-2)$ is undefined
          \end{explanation}


        \item  $f(-5) = $
\begin{explanation}
            $f(-5) = -2$
          \end{explanation}

        \item  $\displaystyle \lim_{x \to 0^+} f(x) = $
\begin{explanation}
            $\displaystyle \lim_{x \to 0^+} f(x) = 0$
          \end{explanation}

        \item $\displaystyle \lim_{x \to 0} f(x) = $
\begin{explanation}
            $\displaystyle \lim_{x \to 0} f(x)$ is undefined
          \end{explanation}

      \end{enumerate}

  \end{enumerate}

\end{problem}
