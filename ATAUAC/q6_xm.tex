\begin{problem}
Consider  a  Riemann sum with $n$ rectangles for the  function $f$ on the interval $[a,b]$.\\
Use geometry to find the limit of Riemann sums as $n\to\infty$.\\
$\lim_{n\to\infty} \sum_{k=1}^{n}  f ( x_k^* ) \Delta x $ ?
\begin{enumerate}
\item $f(x)=x$, $[0,3]$;
	
\item $f(x)=|x|$, $[-3,3]$;
	
\end{enumerate}
\end{problem}
