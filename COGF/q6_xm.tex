\begin{problem}

  Sketch the graph of a function $f$ satisfying all of the conditions:
  \begin{enumerate}
    \item
      $f$ is continuous and \emph{odd}, $f(0) = 0$,

    \item
      $\displaystyle \lim_{x \to \infty} f(x) = -5$,

    \item
      $f'(x) > 0$ on $(6, \infty)$,

    \item
      $f'(x) < 0$ on $(0, 6)$,

    \item
      $f''(x) > 0$ on $(0, 12)$, and
      
    \item
      $f''(x) < 0$ on $(12, \infty)$.
  \end{enumerate}
